final x25519 = X25519();

// 1️⃣ Generate client ephemeral key pair
final clientKeyPair = await x25519.newKeyPair();

// 2️⃣ Server public key (received earlier)
final serverPublicKey = SimplePublicKey(
  serverPubBytes,
  type: KeyPairType.x25519,
);

// 3️⃣ Compute shared secret
final sharedSecret = await x25519.sharedSecretKey(
  keyPair: clientKeyPair,
  remotePublicKey: serverPublicKey,
);

// 4️⃣ Derive AES key
final hkdf = Hkdf(hmac: Hmac.sha256());
final aesKey = await hkdf.deriveKey(
  secretKey: sharedSecret,
  outputLength: 32, // AES-256
);

// 5️⃣ Encrypt
final aes = AesGcm.with256bits();
final nonce = aes.newNonce();

final secretBox = await aes.encrypt(
  messageBytes,
  secretKey: aesKey,
  nonce: nonce,
);

// 6️⃣ Send to server
final clientPublicKeyBytes =
    (await clientKeyPair.extractPublicKey()).bytes;
